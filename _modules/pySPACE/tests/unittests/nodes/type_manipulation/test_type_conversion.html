[docs]    def cmp_TimeSeries(self, t1, t2):
        t1 = t1.view(numpy.ndarray)
        t2 = t2.view(numpy.ndarray)
        for i in range(len(t1)):
            val = self.cmp_value(t1[i][0], t2[i][0])
            if not(val):
                return False
        return True
